006package org.anarres.gradle.plugin.sablecc;
007
008import java.io.File;
009import java.io.IOException;
010import javax.annotation.Nonnull;
011import org.codehaus.groovy.runtime.DefaultGroovyMethods;
012import org.gradle.api.GradleException;
013import org.gradle.api.Task;
014import org.gradle.api.file.ConfigurableFileTree;
015import org.gradle.api.file.EmptyFileVisitor;
016import org.gradle.api.file.FileVisitDetails;
017import org.gradle.api.internal.ConventionTask;
018import org.gradle.api.tasks.InputDirectory;
019import org.gradle.api.tasks.OutputDirectory;
020import org.gradle.api.tasks.TaskAction;
021
022/**
023 *
024 * @author shevek
025 */
026public class SableCC extends ConventionTask {
027
028    private File inputDir;
029    private File outputDir;
030
031    @Nonnull
032    @InputDirectory
033    public File getInputDir() {
034        return inputDir;
035    }
036
037    public void setInputDir(File inputDir) {
038        this.inputDir = inputDir;
039    }
040
041    @Nonnull
042    @OutputDirectory
043    public File getOutputDir() {
044        return outputDir;
045    }
046
047    public void setOutputDir(File outputDir) {
048        this.outputDir = outputDir;
049    }
050
051    @TaskAction
052    public void runSableCC() {
053        // println "Reading from $inputDir"
054        final File outputDir = getOutputDir();
055        // println "Writing to $outputDir"
056        DefaultGroovyMethods.deleteDir(outputDir);
057        outputDir.mkdirs();
058
059        ConfigurableFileTree inputFiles = getProject().fileTree(getInputDir());
060        inputFiles.include("**/*.sablecc");
061        inputFiles.visit(new EmptyFileVisitor() {
062            @Override
063            public void visitFile(FileVisitDetails fvd) {
064                try {
065                    org.sablecc.sablecc.SableCC.processGrammar(fvd.getFile().getAbsolutePath(), outputDir.getAbsolutePath());
066                } catch (Exception e) {
067                    throw new GradleException("Failed to process " + fvd, e);
068                }
069            }
070        });
071
072        ConfigurableFileTree outputFiles = getProject().fileTree(outputDir);
073        outputFiles.visit(new EmptyFileVisitor() {
074            @Override
075            public void visitFile(FileVisitDetails fvd) {
076                try {
077                    SableCCPostProcessor.processFile(fvd.getFile());
078                } catch (IOException e) {
079                    throw new GradleException("Failed to post-process " + fvd, e);
080                }
081            }
082        });
083
084    }
085
086}
