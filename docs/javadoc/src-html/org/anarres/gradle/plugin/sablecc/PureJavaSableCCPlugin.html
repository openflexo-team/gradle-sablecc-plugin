001package org.anarres.gradle.plugin.sablecc;
002
003import groovy.lang.Closure;
004import java.io.File;
005import java.util.ArrayList;
006import java.util.List;
007import java.util.concurrent.Callable;
008import org.anarres.gradle.plugin.velocity.VelocityTask;
009import org.gradle.api.Action;
010import org.gradle.api.Plugin;
011import org.gradle.api.Project;
012import org.gradle.api.Task;
013import org.gradle.api.artifacts.Configuration;
014import org.gradle.api.file.CopySpec;
015import org.gradle.api.plugins.JavaPluginConvention;
016import org.gradle.api.tasks.Copy;
017import org.gradle.api.tasks.SourceSet;
018import org.gradle.api.tasks.SourceSetContainer;
019
020/**
021 *
022 * @author shevek
023 */
024public class PureJavaSableCCPlugin implements Plugin<Project> {
025
026    @Override
027    public void apply(final Project project) {
028        final SableCCPluginExtension extension = project.getExtensions().create("sablecc", SableCCPluginExtension.class);
029        final Configuration configuration = project.getConfigurations().create("sablecc");
030
031        project.getDependencies().add(configuration.getName(), "sablecc:sablecc:3.2-1");
032
033        final Task sableccGrammarTask = project.getTasks().create("sableccGrammar", VelocityTask.class, new Action<VelocityTask>() {
034
035            @Override
036            public void execute(VelocityTask task) {
037
038                task.conventionMapping("inputDir", new Callable<File>() {
039                    @Override
040                    public File call() throws Exception {
041                        return project.file(extension.inputDir);
042                    }
043                });
044
045                task.conventionMapping("includeDirs", new Callable<List<File>>() {
046                    @Override
047                    public List<File> call() {
048                        List<Object> includeDirs = extension.includeDirs;
049                        if (includeDirs == null)
050                            return null;
051                        List<File> out = new ArrayList<File>();
052                        for (Object includeDir : includeDirs)
053                            out.add(project.file(includeDir));
054                        return out;
055                    }
056                });
057
058                task.conventionMapping("outputDir", new Callable<File>() {
059                    @Override
060                    public File call() throws Exception {
061                        return project.file(extension.intermediateDir);
062                    }
063                });
064
065                task.setFilter("**/*.sablecc");
066            }
067        });
068
069        final Task sableccParserTask = project.getTasks().create("sableccParser", SableCC.class, new Action<SableCC>() {
070            @Override
071            public void execute(SableCC task) {
072                task.dependsOn(sableccGrammarTask);
073                task.setDescription("Preprocesses SableCC grammar files.");
074
075                task.conventionMapping("inputDir", new Callable<File>() {
076                    @Override
077                    public File call() throws Exception {
078                        return project.file(extension.intermediateDir);
079                    }
080                });
081
082                task.conventionMapping("outputDir", new Callable<File>() {
083                    @Override
084                    public File call() throws Exception {
085                        return project.file(extension.outputDir);
086                    }
087                });
088            }
089        });
090
091        project.getTasks().getByName("compileJava").dependsOn(sableccParserTask);
092        SourceSetContainer sourceSets = project.getConvention().getPlugin(JavaPluginConvention.class).getSourceSets();
093        final SourceSet mainSourceSet = sourceSets.getByName("main");
094        mainSourceSet.getJava().srcDir(extension.outputDir);
095
096        Task sableccResourcesTask = project.getTasks().create("sableccResources", Copy.class, new Action<Copy>() {
097
098            @Override
099            public void execute(Copy task) {
100                task.setDescription("Copies SableCC resource files.");
101                task.into(mainSourceSet.getOutput().getResourcesDir());
102                task.from(extension.outputDir, new Closure<Void>(PureJavaSableCCPlugin.this) {
103
104                    public Void doCall(CopySpec spec) {
105                        spec.include("**/*.dat");
106                        return null;
107                    }
108                });
109            }
110        });
111        project.getTasks().getByName("classes").dependsOn(sableccResourcesTask);
112
113    }
114
115}
