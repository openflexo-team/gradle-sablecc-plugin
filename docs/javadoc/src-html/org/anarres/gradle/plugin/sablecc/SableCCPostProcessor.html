031    private static List<String> readLines(File file) throws IOException {
032        FileReader fr = new FileReader(file);
033        try {
034            BufferedReader br = new BufferedReader(fr);
035            List<String> out = new ArrayList<String>();
036            for (;;) {
037                String line = br.readLine();
038                if (line == null)
039                    break;
040                out.add(line);
041            }
042            return out;
043        } finally {
044            fr.close();
045        }
046    }

048    private static void writeLines(File file, List<? extends String> lines) throws IOException {
049        StringBuilder buf = new StringBuilder();
050        for (String line : lines)
051            buf.append(line).append("\n");
052        FileWriter fw = new FileWriter(file);
053        fw.write(buf.toString());
054        fw.close();
055    }

057    private static String getBaseName(String name) {
058        name = name.substring(name.lastIndexOf(File.separator) + 1);
059        int index = name.lastIndexOf('.');
060        if (index > 0)
061            name = name.substring(0, index);
062        return name;
063    }

065    public static void processFile(File file) throws IOException {
066        if (LEXER_NAME.equals(file.getName())) {
067            processLexer(file);
068        } else if (PARSER_NAME.equals(file.getName())) {
069            processParser(file);
070        } else if (DEPTH_FIRST_ADAPTER_NAME.equals(file.getName())) {
071            processClone(file);
072        } else if (NODE_NAME.equals(file.getName())) {
073            processNode(file);
074        } else if (TOKEN_NAME.equals(file.getName())) {
075            processToken(file);
076        } else if (ANALYSIS_ADAPTER_NAME.equals(file.getName())) {
077            processAnalysisAdapter(file);
078        } else if (EOF_NAME.equals(file.getName())) {
079            processTToken(file, "EOF");
080        } else if (file.getName().startsWith("T")) {
081            processTToken(file, "T");
082        }
083    }

085    private static void processLexer(File file) throws IOException {
086        @SuppressWarnings({"rawtypes", "unchecked"})
087        List<String> lines = readLines(file);
088        List<String> lines_iface = new ArrayList<String>();
089        for (int i = 0; i < lines.size(); i++) {
090            String line = lines.get(i);
091
092            // Make Lexer implement an interface.
093            if (line.startsWith("package ")) {
094                lines_iface.add(line);
095            } else if (line.startsWith("import ")) {
096                lines_iface.add(line);
097            } else if (line.startsWith("public class Lexer")) {
098                lines.set(i, "public class Lexer implements LexerInterface");
099
100                // Add offsets to Tokens, as well as line/column.
101            } else if (line.contains("private int line")) {
102                lines.add(i++, "    private int offset;");
103            } else if (line.contains("this.line++") || line.contains("this.pos++")) {
104                lines.add(i++, "                this.offset++;");
105            } else if (line.contains("int start_pos = this.pos")) {
106                lines.add(i++, "        int start_offset = this.offset;");
107            } else if (line.contains("int accept_pos = ")) {
108                lines.add(i++, "        int accept_offset = -1;");
109            } else if (line.contains("accept_pos = this.pos")) {
110                lines.add(i++, "            accept_offset = this.offset;");
111            } else if (line.contains("this.pos = accept_pos")) {
112                lines.add(i++, "            this.offset = accept_offset;");
113            } else if (line.contains("this.pos = token.getPos() - 1;")) {
114                lines.add(i++, "        this.offset = token.getOffset();");
115            } else if (line.contains("start_line + 1,")) {
116                lines.add(i++, "                    start_offset,");
117            } else if (line.contains("Token new")) {
118                line = line.replace("int line,", "int offset, @SuppressWarnings(\"hiding\") int line,");
119                line = line.replace("line, pos);", "offset, line, pos);");
120                lines.set(i, line);
121
122                // Improve the performance of text handling.
123            } else if (line.contains("StringBuffer")) {
124                line = line.replace("StringBuffer", "StringBuilder");
125                lines.set(i, line);
126            } else if (line.contains("private String getText(")) {
127                line = "protected String getText(int acceptLength) { return text.substring(0, acceptLength); }";
128                lines.set(i, line);
129                for (int j = 0; j < 9; j++)
130                    lines.remove(i + 1);
131            }
132        }
133        writeLines(file, lines);
134
135        lines_iface.addAll(Arrays.asList(
136                "public interface LexerInterface {",
137                "    public Token peek() throws LexerException, IOException;",
138                "    public Token next() throws LexerException, IOException;",
139                "}"));
140        File ifile = new File(file.getParentFile(), "LexerInterface.java");
141        writeLines(ifile, lines_iface);
142    }

144    private static void processParser(File file) throws IOException {
145        @SuppressWarnings({"rawtypes", "unchecked"})
146        List<String> lines = readLines(file);
147        for (int i = 0; i < lines.size(); i++) {
148            String line = lines.get(i);
149            line = line.replace(" Lexer ", " LexerInterface ");
150            lines.set(i, line);
151        }
152        writeLines(file, lines);
153    }

155    private static String getField(String line) {
156        int start;
157        for (start = 0; start < line.length(); start++)
158            if (Character.isUpperCase(line.charAt(start)))
159                break;
160        int end;
161        for (end = start; end < line.length(); end++)
162            if (!Character.isJavaIdentifierPart(line.charAt(end)))
163                break;
164        return line.substring(start, end - 1);
165    }
166    private static final Pattern GETTER = Pattern.compile("\\s*public\\s+([\\w<>]+)\\s+get(\\w+)\\(\\)");
167    private static final String[] IGNORE = {"Node", "NodeAccessor", "Switch", "Switchable", "Token"};  // Sorted
168
169    private static void processCloneNode(File file, List<String> body) throws IOException {
170        String name = getBaseName(file.getName());
171        if (name.startsWith("P"))
172            return;
173        if (Arrays.binarySearch(IGNORE, name) >= 0)
174            return;
175        body.add("");
176        body.add("\t@Override");
177        body.add("\tpublic void case" + name + "(" + name + " node) {");
178
179        if (name.startsWith("T")) {
180            body.add("\t\tsetClone(node, (" + name + ") node.clone());");
181        } else {
182
183            Map<String, String> fields = new LinkedHashMap<String, String>();
184
185            @SuppressWarnings({"rawtypes", "unchecked"})
186            List<String> lines = readLines(file);
187            for (int i = 0; i < lines.size(); i++) {
188                String line = lines.get(i);
189                Matcher m = GETTER.matcher(line);
190                if (m.find()) {
191                    String type = m.group(1);
192                    type = type.replace("LinkedList<", "List<");
193                    String field = m.group(2);
194                    fields.put(field, type);
195                    body.add("\t\t" + type + " _" + field + "_ = clone(node.get" + field + "());");
196                }
197            }
198
199            StringBuilder buf = new StringBuilder("\t\tsetClone(node, new ");
200            buf.append(name).append('(');
201            boolean b = false;
202            for (String field : fields.keySet()) {
203                if (b)
204                    buf.append(", ");
205                else
206                    b = true;
207                buf.append("_").append(field).append("_");
208            }
209            buf.append("));");
210            body.add(buf.toString());
211        }
212
213        body.add("\t}");
214    }

216    private static void processClone(File file) throws IOException {
217        @SuppressWarnings({"rawtypes", "unchecked"})
218        List<String> lines = readLines(file);
219        List<String> header = new ArrayList<String>();
220        List<String> body = new ArrayList<String>();
221        for (int i = 0; i < lines.size(); i++) {
222            String line = lines.get(i);
223            String trim = line.trim();
224            if (trim.startsWith("package ")) {
225                header.add(line);
226                header.add("");
227            } else if (trim.startsWith("import ")) {
228                header.add(line);
229            }
230        }
231
232        File dir = new File(file.getParent(), "../node");
233        for (File node : dir.listFiles()) {
234            processCloneNode(node, body);
235        }
236
237        List<String> out = new ArrayList<String>();
238        out.addAll(header);
239        out.addAll(Arrays.asList("",
240                "public class CloneVisitor extends AnalysisAdapter {",
241                "",
242                "\tprivate Object clone;",
243                "",
244                "\tprivate <T extends Node> List<T> clone(List<T> in) {",
245                "\t\tif (in == null)",
246                "\t\t\treturn null;",
247                "\t\tList<T> out = new ArrayList<T>(in.size());",
248                "\t\tfor (T node : in)",
249                "\t\t\tout.add(clone(node));",
250                "\t\treturn out;",
251                "\t}",
252                "",
253                "\t@SuppressWarnings({\"unchecked\"})",
254                "\tpublic <T extends Node> T clone(T node) {",
255                "\t\tif (node == null)",
256                "\t\t\treturn null;",
257                "\t\tnode.apply(this);",
258                "\t\treturn (T) clone;",
259                "\t}",
260                "",
261                "\tprivate <T extends Node> void setClone(T prev, T repl) {",
262                "\t\tthis.clone = repl;",
263                "\t\tfireClone(prev, repl);",
264                "\t}",
265                "",
266                "\tprotected <T extends Node> void fireClone(T prev, T repl) {",
267                "\t}",
268                "",
269                "\t@Override",
270                "\tpublic void defaultCase(Node node) {",
271                "\t\tthrow new IllegalStateException();",
272                "\t}"));
273        out.addAll(body);
274        out.add("}");
275        for (int i = 0; i < out.size(); i++)
276            out.set(i, out.get(i).replace("\t", "    "));
277        File cfile = new File(file.getParentFile(), "CloneVisitor.java");
278        writeLines(cfile, out);
279
280    }

282    private static void processNode(File file) throws IOException {
283        @SuppressWarnings({"rawtypes", "unchecked"})
284        List<String> lines = readLines(file);
285        List<String> header = new ArrayList<String>();
286        for (int i = 0; i < lines.size(); i++) {
287            String line = lines.get(i);
288            String trim = line.trim();
289            if (trim.startsWith("package ")) {
290                header.add(line);
291            }
292        }
293
294        List<String> out = new ArrayList<String>(header);
295        out.addAll(Arrays.asList("",
296                "import javax.annotation.CheckForNull;",
297                "import javax.annotation.Nonnull;",
298                "",
299                "public class NodeAccessor {",
300                "",
301                "\tprivate NodeAccessor() {",
302                "\t}",
303                "",
304                "\tpublic static void setParent(@Nonnull Node node, @CheckForNull Node parent) {",
305                "\t\tnode.parent(parent);",
306                "\t}",
307                "",
308                "\tpublic static void removeChild(@Nonnull Node parent, @Nonnull Node child) {",
309                "\t\tparent.removeChild(child);",
310                "\t}",
311                "}"));
312        for (int i = 0; i < out.size(); i++)
313            out.set(i, out.get(i).replace("\t", "    "));
314        File cfile = new File(file.getParentFile(), "NodeAccessor.java");
315        writeLines(cfile, out);
316    }

318    private static void processToken(File file) throws IOException {
319        @SuppressWarnings({"rawtypes", "unchecked"})
320        List<String> lines = readLines(file);
321        for (int i = 0; i < lines.size(); i++) {
322            String line = lines.get(i);
323            if (line.contains("private int line"))
324                lines.add(i++, "    private int offset;");
325            else if (line.contains("public int getLine"))
326                lines.add(i++, "    public int getOffset() { return offset; }\n    public void setOffset(int offset) { this.offset = offset; }\n");
327            // lines.set(i, line);
328        }
329        writeLines(file, lines);
330    }

332    private static void processTToken(File file, String prefix) throws IOException {
333        @SuppressWarnings({"rawtypes", "unchecked"})
334        List<String> lines = readLines(file);
335        for (int i = 0; i < lines.size(); i++) {
336            String line = lines.get(i);
337            if (line.contains("int line, int pos)")) {
338                line = line.replace("int line, int pos", "int offset, int line, int pos");
339                lines.set(i, line);
340            } else if (line.contains("setLine(line)")) {
341                lines.add(i++, "        setOffset(offset);");
342            } else if (line.contains("return new " + prefix)) {
343                line = line.replace("getLine(), getPos()", "getOffset(), getLine(), getPos()");
344                lines.set(i, line);
345            }
346        }
347        writeLines(file, lines);
348    }

350    private static void processAnalysisAdapter(File file) throws IOException {
351        @SuppressWarnings({"rawtypes", "unchecked"})
352        List<String> lines = readLines(file);
353        for (int i = 0; i < lines.size(); i++) {
354            String line = lines.get(i);
355            line = line.replace("private Hashtable", "private Map");
356            line = line.replace("new Hashtable", "new WeakHashMap");
357            line = line.replace("public void case", "@Override\n    public void case");
358            line = line.replace("public Object get", "@Override\n    public Object get");
359            line = line.replace("public void set", "@Override\n    public void set");
360            lines.set(i, line);
361        }
362        writeLines(file, lines);
363    }
